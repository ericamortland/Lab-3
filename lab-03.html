pop_url <- 'https://www2.census.gov/programs-surveys/popest/datasets/2020-2023/counties/totals/co-est2023-alldata.csv'
pop_data <- read.csv(pop_url)

pop_data <- pop_data %>%
  mutate(
    STATE_FIP = str_pad(as.character(STATE), width = 2, side = "left", pad = "0"),
    COUNTY_FIP = str_pad(as.character(COUNTY), width = 3, side = "left", pad = "0"),
    FIP = paste(STATE_FIP, COUNTY_FIP, sep = "")
  )
pop_data <- pop_data %>%
  select(contains("NAME"), contains("2021")) %>%
  filter('COUNTY_FIP' != "000")
